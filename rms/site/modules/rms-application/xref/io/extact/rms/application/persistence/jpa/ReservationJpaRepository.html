1   package io.extact.rms.application.persistence.jpa;

3   import java.time.LocalDate;
4   import java.time.LocalDateTime;
5   import java.util.List;
6   import java.util.stream.Collectors;
7   
8   import javax.enterprise.context.ApplicationScoped;
9   import javax.persistence.EntityManager;
10  import javax.persistence.PersistenceContext;
11  
12  import io.extact.rms.application.domain.Reservation;
13  import io.extact.rms.application.persistence.ReservationRepository;
14  import io.extact.rms.application.persistence.GenericRepository.ApiType;
15  import io.extact.rms.platform.extension.EnabledIfRuntimeConfig;
16  
17  @ApplicationScoped
18  @EnabledIfRuntimeConfig(propertyName = ApiType.PROP_NAME, value = ApiType.JPA)
19  public class ReservationJpaRepository implements ReservationRepository, JpaCrudRepository<Reservation> {
20  
21      private static final String JPQL_SELECT_BY_RENTAL_ID = "select r from Reservation r where r.rentalItemId = ?1 order by r.id";
22  
23      @PersistenceContext
24      private EntityManager em;
25  
26      @Override
27      public List<Reservation> findByRentalItemAndStartDate(int rentalItemId, LocalDate startDate) {
28          return em.createQuery(JPQL_SELECT_BY_RENTAL_ID, Reservation.class)
29                      .setParameter(1, rentalItemId)
30                      .getResultList().stream()
31                      .filter(reservation -> reservation.getStartDateTime().toLocalDate().equals(startDate))
32                      .collect(Collectors.toList());
33      }
34  
35      @Override
36      public List<Reservation> findByReserverId(int reserverId) {
37          var jpql = "select r from Reservation r where r.userAccountId = ?1 order by r.id";
38          return em.createQuery(jpql, Reservation.class)
39                      .setParameter(1, reserverId)
40                      .getResultList();
41      }
42  
43      @Override
44      public List<Reservation> findByRentalItemId(int rentalItemId) {
45          return em.createQuery(JPQL_SELECT_BY_RENTAL_ID, Reservation.class)
46                      .setParameter(1, rentalItemId)
47                      .getResultList();
48      }
49  
50      @Override
51      public Reservation findOverlappedReservation(int rentalItemId, LocalDateTime startDateTime, LocalDateTime endDateTime) {
52          var period = new Reservation.DateTimePeriod(startDateTime, endDateTime);
53          return em.createQuery(JPQL_SELECT_BY_RENTAL_ID, Reservation.class)
54                      .setParameter(1, rentalItemId)
55                      .getResultList().stream()
56                      .filter(reservation -> reservation.getReservePeriod().isOverlappedBy(period))
57                      .findFirst()
58                      .orElse(null);
59      }
60  
61      @Override
62      public List<Reservation> findOverlappedReservation(LocalDateTime from, LocalDateTime to) {
63          var period = new Reservation.DateTimePeriod(from, to);
64          return findAll().stream()
65                  .filter(reservation -> reservation.getReservePeriod().isOverlappedBy(period))
66                  .collect(Collectors.toList());
67      }
68  
69      @Override
70      public EntityManager getEntityManage() {
71          return this.em;
72      }
73  
74      @Override
75      public Class<Reservation> getTargetClass() {
76          return Reservation.class;
77      }
78  }
