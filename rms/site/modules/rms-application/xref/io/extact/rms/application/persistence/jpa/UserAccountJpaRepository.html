8   import io.extact.rms.application.domain.UserAccount;
9   import io.extact.rms.application.persistence.UserAccountRepository;
10  import io.extact.rms.application.persistence.GenericRepository.ApiType;

12  
13  @ApplicationScoped
14  @EnabledIfRuntimeConfig(propertyName = ApiType.PROP_NAME, value = ApiType.JPA)
15  public class UserAccountJpaRepository implements UserAccountRepository, JpaCrudRepository<UserAccount> {
16  
17      @PersistenceContext
18      private EntityManager em;
19  
20      @Override
21      public UserAccount findByLoginIdAndPasswod(String loginId, String password) {
22          var jpql = "select u from UserAccount u where u.loginId = ?1 and u.password = ?2";
23          try {
24              return em.createQuery(jpql, UserAccount.class)
25                          .setParameter(1, loginId)
26                          .setParameter(2, password)
27                          .getSingleResult();
28          } catch (NoResultException e) {
29              return null;
30          }
31      }
32  
33      @Override
34      public UserAccount findByLoginId(String loginId) {
35          var jpql = "select u from UserAccount u where u.loginId = ?1";
36          try {
37              return em.createQuery(jpql, UserAccount.class)
38                          .setParameter(1, loginId)
39                          .getSingleResult();
40          } catch (NoResultException e) {
41              return null;
42          }
43      }
44  
45      @Override
46      public EntityManager getEntityManage() {
47          return this.em;
48      }
49  
50      @Override
51      public Class<UserAccount> getTargetClass() {
52          return UserAccount.class;
53      }
54  }
