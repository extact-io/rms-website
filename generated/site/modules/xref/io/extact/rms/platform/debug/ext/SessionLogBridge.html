1   package io.extact.rms.platform.debug.ext;
2   
3   import org.eclipse.persistence.platform.server.ServerLog;
4   
5   import lombok.extern.slf4j.Slf4j;
6   
7   @Slf4j(topic = "EclipseLink")
8   public class SessionLogBridge extends ServerLog {
9   
10      // EclipseLink側でDateTimeがappendされないようにする
11      @Override
12      public boolean shouldPrintDate() {
13          return false;
14      }
15  
16      @Override
17      protected void basicLog(int level, String category, String message) {
18          switch (level) {
19              case OFF, SEVERE        -> log.error(message);
20              case WARNING            -> log.warn(message);
21              case INFO               -> log.info(message);
22              case CONFIG, FINE       -> log.debug(message);
23              case FINER, FINEST, ALL -> log.trace(message);
24              default                 -> log.trace(message);
25          };
26      }
27  }
